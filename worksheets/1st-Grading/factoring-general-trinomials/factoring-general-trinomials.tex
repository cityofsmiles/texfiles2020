    		\item $ 15k^2 - 27k - 6 = 3 \left(k - 2\right) \left(5 k + 1\right)$
    		\item $ 4n^2 - 15n - 25 = \left(n - 5\right) \left(4 n + 5\right)$
    		\item $ 4b^2 - 17b + 4 = \left(b - 4\right) \left(4 b - 1\right)$
    		\item $ 6n^2 + 37n + 6 = \left(n + 6\right) \left(6 n + 1\right)$
    		\item $ 6k^2 + 5k - 6 = \left(2 k + 3\right) \left(3 k - 2\right)$
    		\item $ 2p^2 + 2p - 4 = 2 \left(p - 1\right) \left(p + 2\right)$
