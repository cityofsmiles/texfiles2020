    \noindent \textbf{A. Factoring Integers} \\
    
    Completely factor the following numbers. One point each.
    
    \begin{enumerate}
    	\begin{multicols}{2}
    	\item 28
    	\item 36
    	\item 60
    	\item 45
    	\item 54
        \end{multicols}
    \end{enumerate}

    \vspace*{1.3ex}

    \noindent \textbf{B. Finding the Common Factor} \\
    
    Find the greatest common monomial factor of each polynomial. One point each.
    
    \begin{enumerate}
    	\begin{multicols}{2}
    	\item $ 6x^2 + 3xy $
    	\item $ 24abx + 20ab $
    	\item $ 15x^2y - 6xy $
    	\item $ 36ab + 60b $
    	\item $ 18a^3 - 24a^2 $
    	\item $ 12x^2y + 12xy^2 + 24 xyz $
    	\item $ 3x^4 - 6x^3y $
    	\item $ 9x^2 + 9xy $
    	\item $ 16mnx + 28mn $
    	\item $ 15x^2 +20x $
        \end{multicols}
    \end{enumerate}

    \vspace*{1.3ex}
    
    \noindent \textbf{C. Factoring Polynomials with Common Monomial Factor} \\
    
    Factor each polynomial completely. One point each.
    
    \begin{enumerate}
    	\begin{multicols}{2}
    		\item $ 6x^2 + 18x^3y $
    		\item $ 14a^2bx + 28ab $
    		\item $ 25x^2y - 60xy^2 $
    		\item $ 26ab^2 + 52b^3 $
    		\item $ 18a^2 - 72a^4 $
    		\item $ 22x^3y + 11x^2y^2 + 44 xyz $
    		\item $ 32x^2 - 64x^3y $
    		\item $ 90x^2y^3 + 18x^3y $
    		\item $ 36mn^2x + 81mn^3 $
    		\item $ 25x^2y +30xy^2 $
    	\end{multicols}
    \end{enumerate}

    	 \noindent \textbf{A. Factoring Integers} \\
    	 
    	 \begin{enumerate}
    	 	\begin{multicols}{2}
    	 		\item $ 28 = (2^2)(7)$
    	 		\item $ 36 = (2^2)(3^2)$
    	 		\item $ 60 = (2^2)(3)(5)$
    	 		\item $ 45 = (3^2)(5)$
    	 		\item $ 54 = (2)(3^3)$
    	 	\end{multicols}
    	 \end{enumerate}
    	 
    	 \vspace*{1.3ex}
    	 
    	 \noindent \textbf{B. Finding the Common Factor} \\
    	     	 
    	 \begin{enumerate}
    	 	\begin{multicols}{2}

    	 		\item $ 4ab $
    	 		\item $ 3xy $
    	 		\item $ 12b $
    	 		\item $ 6a^2 $
    	 		\item $ 12xy $
    	 		\item $ 3x^3 $
    	 		\item $ 9x $
    	 		\item $ 4mn $
    	 		\item $ 5x $
    	 	\end{multicols}
    	 \end{enumerate}
