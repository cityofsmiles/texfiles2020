    \noindent \textbf{A. Finding the Cube Root} \\
    
    Express the following in exponential form with a power of 3. One point each.

    	\item $ \frac{1}{8} $

    \end{enumerate}

    \vspace*{2em}
    
    \noindent \textbf{A. Finding the Cube Root} \\
        
    \begin{enumerate}
    	\begin{multicols}{2}
    		\item $ 125 = 5^3$ 
    		\item $ \frac{1}{8} = (\frac{1}{2})^3$
    		\item $ -8 = (-2)^3$
    		\item $ 216y^3 = (6y)^3$
    		\item $ -64x^6y^{12} = (-4x^2y^4)^3$
    	\end{multicols}
    \end{enumerate}
    
    \vspace*{1.3ex}
    
    \noindent \textbf{B. Factoring the Sum and Difference of Two Cubes} \\
        
    \begin{enumerate}
    	%\begin{multicols}{2}
    		\item $ x^3 + 125 = \left(x + 5\right) \left(x^{2} - 5 x + 25\right)$
    		\item $ m^3 - 64 = \left(m - 4\right) \left(m^{2} + 4 m + 16\right)$
    		\item $ 8x^3 - 27 =\left(2 x - 3\right) \left(4 x^{2} + 6 x + 9\right) $
    		\item $ 1 - a^3 = \left(1-a\right) \left(1 + a + a^{2}\right) $
    		\item $ n^3 + 27 = \left(n + 3\right) \left(n^{2} - 3 n + 9\right)$
    		\item $ 27x^3 - 8 =\left(3 x - 2\right) \left(9 x^{2} + 6 x + 4\right) $
    		\item $ 64n^3 + 1 = \left(4 n + 1\right) \left(16 n^{2} - 4 n + 1\right)$
    		\item $ 343m^3 + 64n^3 = \left(7 m + 4 n\right) \left(49 m^{2} - 28 m n + 16 n^{2}\right)$
    		\item $ a^3 - 343b^3 = \left(a - 7 b\right) \left(a^{2} + 7 a b + 49 b^{2}\right)$
    		\item $ 16x^4 + 54xy^3 = 2 x \left(2 x + 3 y\right) \left(4 x^{2} - 6 x y + 9 y^{2}\right)$
    	%\end{multicols}
    \end{enumerate}
