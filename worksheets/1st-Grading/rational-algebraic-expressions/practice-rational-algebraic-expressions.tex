    	\item $ \dfrac{3}{a-4} $

    \end{enumerate}

    \noindent \textbf{B. Finding Domains} \\
    
    Find the domain of each rational algebraic expression. Write the final answers only. One point each.
    
    \begin{enumerate}
    	\begin{multicols}{2}
    	\item $ \dfrac{5x}{3x-2} $
    	\item $ \dfrac{7}{3y-6} $
    	\item $ \dfrac{4+a}{a^2 + 2a} $
    	\item $ \dfrac{2y^2 + 13 y + 15}{y + 1} $
    	\item $ \dfrac{y+1}{y-3} $
    	\item $ \dfrac{9}{x^2 + x - 12} $
    	\item $ \dfrac{2 + 4c}{c + 5} $
    	\item $ \dfrac{7}{x^2 + 5x + 6} $
    	\item $ \dfrac{3x-1}{9} $
    	\item $ 2x^3 + 3x^2 + x + 1 $
        \end{multicols}
    \end{enumerate}

    \vspace*{1.3ex}
    
    \noindent \textbf{C. Becoming Meaningless} \\
    
        Find the value or values for which each rational algebraic expression becomes undefined. Write the final answers only. One point each.
    
    \begin{enumerate}
    	\begin{multicols}{2}
    		\item $ \dfrac{3x^2 - 2x + 1}{5x} $
    		\item $ \dfrac{x^2 + 1}{x^2 + 5x + 6} $
    		\item $ \dfrac{x^2 + x + 1}{5x + 10} $
    		\item $ \dfrac{x + 5}{2x} $
    		\item $ \dfrac{x^2 - 1}{x^2 + x - 12} $
    		\item $ \dfrac{k-1}{k+1} $
    		\item $ \dfrac{a^2}{2a + 8} $
    		\item $ \dfrac{10 + v}{w^2 + 4w} $
    		\item $ \dfrac{3m}{2m - 1} $
    		\item $ \dfrac{x - 1}{x^2 - 4} $
    	\end{multicols}
    \end{enumerate}
