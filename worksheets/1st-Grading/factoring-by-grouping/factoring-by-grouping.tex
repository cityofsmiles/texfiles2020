    \noindent \textbf{A. Finding the Common Factor} \\
    
    Find the greatest common monomial factor of each polynomial. One point each.
    
    \begin{enumerate}
    	\begin{multicols}{2}
    	\item $ 6x^2 + 28x^3y $
    	\item $ 21a^2bx + 28ab $
    	\item $ 20x^2y - 60xy^2 $
    	\item $ 24ab^2 + 42b^3 $
    	\item $ 18a^2 - 27a^4 $
        \end{multicols}
    \end{enumerate}

    \vspace*{1.3ex}

    \noindent \textbf{B. Factoring by Grouping} \\
    
    Factor each polynomial completely. Write the final answers only. One point each.
    
    \begin{enumerate}
    	\begin{multicols}{2}
    	\item $ 8r^3 - 64r^2 + r -8 $
    	\item $ 12x^3 + 2x^2 - 30x - 5 $
    	\item $ 63n^3 + 54n^2 - 105n -90 $
    	\item $ 25v^3 + 5v^2 + 30v + 6 $
    	\item $ 96n^3 - 84n^2 + 112n - 98 $
    	\item $ 4v^3 - 12v^2 - 5v + 15 $
    	\item $ 24p^3 + 15p^2 - 56p - 35 $
    	\item $ 56xw + 49xk^2 - 24yw - 21yk^2 $
    	\item $ 12x^2u + 3x^2v + 28yu + 7yv $
    	\item $ 12bc - 4bd - 15xc + 5xd $
        \end{multicols}
    \end{enumerate}

    \vspace*{1.3ex}

    		\item $ 4v^3 - 12v^2 - 5v + 15 = \left(v - 3\right) \left(4 v^{2} - 5\right)
    		$
