    \noindent \textbf{A. True or False} 
    
    Write True if the expression is a rational algebraic expression or False if it is not. One point each.
    
    \begin{enumerate}
    	\begin{multicols}{2}
    	\item $ \dfrac{4x}{x+8} $
    	\item $ \dfrac{3\sqrt{x}+1}{19} $
    	\item $ \dfrac{x^2 - 4}{x^{\frac{1}{3}}} $
    	\item $ \dfrac{x-1}{\sqrt{2x} + 3} $
    	\item $ \dfrac{3}{a-3} $
        \end{multicols}
    \end{enumerate}

    \noindent \textbf{C. Multiplying and Dividing Rational Expressions} 
    
    Perform the indicated operation. Choose the answer from the box. Write the letter only. One point each.
    
    \begin{center} \fbox{\fbox{\parbox{5.5in}{\centering \begin{enumerate}[label=\alph*. ]
    					\begin{multicols}{2}
    						\item  $ \dfrac{4 \left(x - 8\right)}{x + 2} $
    						\item  $ \dfrac{2}{3 x \left(x - 5\right)} $
    						\item  $ \dfrac{3}{2 x \left(x + 4\right)} $	
    						\item  $ \dfrac{7}{x - 7} $
    						\item  $ \dfrac{5}{x + 6} $	
    						\item  $ \dfrac{x + 3}{x - 10} $
    						\item  $ 2 \left(x - 1\right) $		
    						\item  $ \dfrac{8 x \left(x - 2\right)}{x - 9} $
    						\item  $ \dfrac{10 x^{2}}{x + 2} $
    						\item  $ \dfrac{x + 10}{3} $
    					\end{multicols}
    	\end{enumerate}}}} 
    \end{center}
    
    \begin{enumerate}[resume]
    	\begin{multicols}{2}
    		\item  $ \dfrac{x^2 - 1}{x} \cdot \dfrac{2x}{x + 1} $			
    		\item  $ \dfrac{1}{x + 5} \cdot \dfrac{7x + 35}{x - 7} $

    		\item $ \dfrac{3}{2x} \div \dfrac{7x + 28}{7} $						 	
    		\item $ \dfrac{12x + 12}{x - 9} \div \dfrac{12x + 12}{8x^2 - 16x} $					 	
    		\item $ \dfrac{5}{7x^2} \div \dfrac{x^2 + 12x + 36}{7x^3 + 42x^2} $						 	
    		\item $ \dfrac{10x^3 + 100x^2}{6x^2 - 48x} \div \dfrac{x^2 + 12x + 20}{6x^2 - 48x} $				 	
    		\item $ \dfrac{x + 3}{x^2 -x - 90} \div \dfrac{x - 1}{x^2 + 8x -9} $				
    	\end{multicols}
    \end{enumerate}
